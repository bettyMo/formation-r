Les personnages tueurs

perso_fr<-select(perso_fr,nom,sexe,maison, mort,Tue,Tue_par)
killer<-perso_fr %>% 
         count(Tue_par, sort = TRUE) %>%
         filter(!is.na(Tue_par))%>%
         right_join(perso_fr,by=c('Tue_par'='nom'))%>%
          rename(nom=Tue_par,
                 personnes_tue_n=n)
filter(killer,nom=="Jon Snow")
